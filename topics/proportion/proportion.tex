

\section*{1. Core Concept}
\begin{itemize}
    \item Understanding ratios and proportions
    \item Solving missing values using cross multiplication
    \item Recognizing direct vs inverse relationships
    \item Word problems involving proportions
\end{itemize}

\section*{2. Key Rules}

\subsection*{Definition of Proportion}
\begin{align*}
\frac{a}{b} = \frac{c}{d} \Rightarrow ad = bc
\end{align*}

\subsection*{Direct Proportion}
\begin{align*}
y = kx
\end{align*}

\subsection*{Inverse Proportion}
\begin{align*}
xy = k \quad \text{or} \quad y = \frac{k}{x}
\end{align*}

\section*{3. Common SAT Traps}
\begin{itemize}
    \item Mixing up corresponding values
    \item Incorrect cross multiplication
    \item Not simplifying ratios first
    \item Confusing direct and inverse proportion
    \item Ignoring units in word problems
\end{itemize}

\section*{4. Worked Examples}

\subsection*{Example 1}
Solve:
\[
\frac{x}{4} = \frac{3}{5}
\]
\[
5x = 12 \Rightarrow x = \frac{12}{5}
\]

\subsection*{Example 2}
If 3 pens cost 6 dollars, how much do 5 pens cost?
\[
\frac{3}{6} = \frac{5}{x} \Rightarrow 3x = 30 \Rightarrow x = 10
\]

\subsection*{Example 3 (Inverse)}
If 4 workers complete a job in 6 days, how long for 8 workers?
\[
4 \times 6 = 8 \times x \Rightarrow 24 = 8x \Rightarrow x = 3
\]

\section*{5. Practice Questions}

\subsection*{Easy}
\begin{enumerate}
    \item $\frac{x}{3} = \frac{4}{9}$
    \item Simplify the ratio $10:20$
\end{enumerate}

\subsection*{Medium}
\begin{enumerate}
    \setcounter{enumi}{2}
    \item If 2 books cost 8 dollars, what is the cost of 5 books?
    \item $\frac{a}{b} = \frac{6}{10}$, find $a$ when $b = 5$
\end{enumerate}

\subsection*{Hard}
\begin{enumerate}
    \setcounter{enumi}{4}
    \item 6 workers finish a task in 8 days. How long for 12 workers?
    \item A map scale is $1:50$. If a distance is 10 cm on the map, what is the actual distance?
\end{enumerate}

\section*{Answers}
\begin{enumerate}
    \item $x = \frac{4}{3}$
    \item $1:2$
    \item $20$
    \item $3$
    \item $4$ days
    \item $500$ cm
\end{enumerate}

\section*{6. Teaching Tips}
\begin{itemize}
    \item Align ratios clearly (write them vertically)
    \item Match like quantities (units matter)
    \item Identify direct vs inverse relationships
    \item Simplify ratios before solving
\end{itemize}

\section*{7. Speed Strategy}
\begin{itemize}
    \item Cross multiply immediately
    \item Use mental scaling when possible
    \item Apply unit rate for quick solutions
    \item For inverse proportion, keep product constant
\end{itemize}
