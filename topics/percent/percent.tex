

\section*{Core Concept}
\begin{itemize}
    \item Converting between percent, decimal, and fraction
    \item Finding percent of a number
    \item Percent increase and decrease
    \item Reverse percent problems (finding original value)
    \item Multi-step percent changes
\end{itemize}

\section*{Key Rules}

\subsection*{Basic Percent Definition}
\begin{align*}
\text{Percent} = \frac{\text{Part}}{\text{Whole}} \times 100
\end{align*}

\subsection*{Percent of a Number}
\begin{align*}
\text{Part} = \frac{p}{100} \times \text{Whole}
\end{align*}

\subsection*{Percent Change}
\begin{align*}
\text{Percent Change} = \frac{\text{New} - \text{Original}}{\text{Original}} \times 100
\end{align*}

\section*{Common SAT Traps}
\begin{itemize}
    \item Using the wrong base (original vs new value)
    \item Forgetting to convert percent to decimal
    \item Adding percentages incorrectly in multi-step problems
    \item Not recognizing reverse percent questions
    \item Assuming 20\% increase then 20\% decrease returns original value (incorrect)
\end{itemize}

\section*{Worked Examples}

\subsection*{Example 1}
Find 20\% of 150
\[
0.2 \times 150 = 30
\]

\subsection*{Example 2}
A price increases from 50 to 65. Find percent increase
\[
\frac{65 - 50}{50} \times 100 = 30\%
\]

\subsection*{Example 3 (Reverse Percent)}
After a 20\% discount, price is 80. Find original price
\[
0.8x = 80 \Rightarrow x = 100
\]

\subsection*{(Successive Change)}
Increase 100 by 20\%, then decrease by 20\%
\[
100 \rightarrow 120 \rightarrow 96
\]

\section*{Practice Questions}

\subsection*{Easy}
\begin{enumerate}
    \item 25\% of 200
    \item Convert 40\% to decimal
\end{enumerate}

\subsection*{Medium}
\begin{enumerate}
    \setcounter{enumi}{2}
    \item A value increases from 80 to 100. Find percent increase
    \item 15 is what percent of 60?
\end{enumerate}

\subsection*{Hard}
\begin{enumerate}
    \setcounter{enumi}{4}
    \item After a 25\% increase, value is 125. Find original value
    \item A price decreases by 10\% then increases by 10\%. What is the final effect?
\end{enumerate}

\section*{Answers}
\begin{enumerate}
    \item 50
    \item 0.4
    \item 25\%
    \item 25\%
    \item 100
    \item 99\% of original (1\% decrease)
\end{enumerate}

\section*{6. Teaching Tips}
\begin{itemize}
    \item Emphasize that percent means ``out of 100''
    \item Always identify the base (original value)
    \item Use simple numbers like 100 for intuition
    \item Teach reverse percent thinking
\end{itemize}

\section*{Speed Strategy}
\begin{itemize}
    \item Convert percent to decimal immediately
    \item Use 100 as a base for quick reasoning
    \item For successive changes, multiply factors (e.g., $1.2 \times 0.8$)
    \item Avoid unnecessary algebra when mental math works
\end{itemize}
