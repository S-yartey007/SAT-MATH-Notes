
\section*{Core Concept}
\begin{itemize}
    \item Simplifying expressions with exponents and roots
    \item Rewriting between radical form and exponent form
    \item Solving equations with exponents
    \item Handling fractional and negative exponents
    \item Rationalizing denominators
\end{itemize}

\section*{Key Rules}

\subsection*{Exponent Laws}
\begin{align*}
a^m \cdot a^n &= a^{m+n} \\
\frac{a^m}{a^n} &= a^{m-n} \\
(a^m)^n &= a^{mn} \\
a^0 &= 1 \\
a^{-n} &= \frac{1}{a^n}
\end{align*}

\subsection*{Fractional Exponents}
\begin{align*}
a^{\frac{m}{n}} = \sqrt[n]{a^m} = \left(\sqrt[n]{a}\right)^m
\end{align*}

\section*{Common SAT Traps}
\begin{itemize}
    \item Thinking $a^0 = 0$ (Incorrect; $a^0 = 1$)
    \item Forgetting negative exponents mean reciprocals
    \item $\sqrt{a + b} \ne \sqrt{a} + \sqrt{b}$
    \item $\sqrt{x^2} = |x|$, not just $x$
    \item Not simplifying completely
\end{itemize}

\section*{Worked Examples}

\subsection*{Example 1}
\[
\frac{2^3 \cdot 2^5}{2^4} = 2^{3+5-4} = 2^4 = 16
\]

\subsection*{Example 2}
\[
x^{3/2} = \sqrt{x^3} = x\sqrt{x}
\]

\subsection*{Example 3}
\[
\frac{1}{x^{-2}} = x^2
\]

\subsection*{Example 4}
\[
\frac{3}{\sqrt{5}} = \frac{3\sqrt{5}}{5}
\]

\section*{Practice Questions}

\subsection*{Easy}
\begin{enumerate}
    \item $5^2 \cdot 5^3$
    \item Rewrite $x^{-3}$ without negative exponent
\end{enumerate}

\subsection*{Medium}
\begin{enumerate}
    \setcounter{enumi}{2}
    \item $(2^3)^4$
    \item $16^{1/2}$
\end{enumerate}

\subsection*{Hard}
\begin{enumerate}
    \setcounter{enumi}{4}
    \item $\frac{x^{5/2}}{x^{1/2}}$
    \item $\frac{4}{2\sqrt{3}}$
\end{enumerate}

\section*{Answers}
\begin{enumerate}
    \item $5^5 = 3125$
    \item $\frac{1}{x^3}$
    \item $2^{12} = 4096$
    \item $4$
    \item $x^2$
    \item $\frac{2\sqrt{3}}{3}$
\end{enumerate}

\section*{Teaching Tips}
\begin{itemize}
    \item Emphasize pattern recognition (same base rules)
    \item Practice converting radicals and exponents both ways
    \item Encourage mental simplification
    \item Always simplify completely
\end{itemize}

\section*{Speed Strategy}
\begin{itemize}
    \item Combine exponents in one step
    \item Convert radicals to exponents if messy
    \item Cancel early in fractions
    \item Memorize perfect squares and cubes
\end{itemize}
